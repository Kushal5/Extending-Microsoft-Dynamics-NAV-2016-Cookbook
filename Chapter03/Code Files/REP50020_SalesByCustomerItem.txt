    VAR
      TempValueEntry@1000 : TEMPORARY Record 5802;
      EntryNo@1001 : Integer;

    LOCAL PROCEDURE InitTempValueEntry@1(ItemNo@1000 : Code[20]) EntryExists : Boolean;
    BEGIN
      TempValueEntry.INIT;
      TempValueEntry.SETRANGE("Item No.",ItemNo);
      EntryExists := TempValueEntry.FINDFIRST;

      IF NOT EntryExists THEN BEGIN
        EntryNo += 1;
        TempValueEntry."Entry No." := EntryNo;
      END;
    END;

    LOCAL PROCEDURE UpdateTempValueEntry@2(ItemLedgerEntry@1001 : Record 32);
    VAR
      Item@1000 : Record 27;
    BEGIN
      Item.GET(ItemLedgerEntry."Item No.");

      TempValueEntry."Source No." := ItemLedgerEntry."Source No.";
      TempValueEntry."Item No." := ItemLedgerEntry."Item No.";
      TempValueEntry.Description := Item.Description;
      TempValueEntry."Item Ledger Entry Quantity" += ItemLedgerEntry.Quantity;
      TempValueEntry."Cost Amount (Actual)" += ItemLedgerEntry."Cost Amount (Actual)";
      TempValueEntry."Sales Amount (Actual)" += ItemLedgerEntry."Sales Amount (Actual)";
    END;

    BEGIN
    END.
